模板①

public static int ex(int a, int n){
    if(n==1){
        return a;
    }
    int temp = a;
    int res = 1;
    int exponent = 1;
    while((exponent << 1) < n){
        temp *= temp;
        exponent = exponent<<1;
    }
    res *= ex(a,exponent); 
    return res*temp;

}
